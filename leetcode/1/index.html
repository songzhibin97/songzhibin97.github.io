 func twoSum(nums []int, target int) []int {
     for i := 0; i < len(nums); i++ {
         for j := i+1; j < len(nums); j++ {
             if nums[i] + nums[j] == target {
                 return []int{i,j}
             }
         }
     }
     return nil
 }

func twoSum(nums []int, target int) []int {
    m := make(map[int]int)
    for i := 0; i < len(nums); i++ {
        if v,ok := m[target-nums[i]]; ok {
            return []int{v,i}
        }
        m[nums[i]] = i
    }
    return nil
}

func twoSum(nums []int, target int) []int {
    sort.Ints(nums)
    i, j := 0, len(nums)-1
    for i < j {
        if nums[i] + nums[j] == target {
            return []int{i,j}
        } else if nums[i] + nums[j] < target {
            i++
        } else {
            j--
        }
    }
    return nil
}
